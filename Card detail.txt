    <script>
        const card = document.getElementById('card');
        const cardNumber = document.getElementById('cardNumber');
        const cardName = document.getElementById('cardName');
        const expiry = document.getElementById('expiry');
        const cvv = document.getElementById('cvv');
        const displayNumber = document.getElementById('displayNumber');
        const displayName = document.getElementById('displayName');
        const displayExpiry = document.getElementById('displayExpiry');
        const displayCVV = document.getElementById('displayCVV');

        // Card number formatting and display
        cardNumber.addEventListener('input', (e) => {
            let value = e.target.value.replace(/\s/g, '');
            let formattedValue = value.match(/.{1,4}/g)?.join(' ') || value;
            e.target.value = formattedValue;
            
            if (value.length > 0) {
                // Show first 4 and last 4 digits, mask the middle
                let maskedValue = '';
                if (value.length <= 4) {
                    maskedValue = value.padEnd(4, '#');
                } else if (value.length <= 8) {
                    maskedValue = value.slice(0, 4) + ' ' + '****';
                } else if (value.length <= 12) {
                    maskedValue = value.slice(0, 4) + ' **** ****';
                } else {
                    maskedValue = value.slice(0, 4) + ' **** **** ' + value.slice(12);
                }
                displayNumber.textContent = maskedValue;
            } else {
                displayNumber.textContent = '#### #### #### ####';
            }
        });

        // Card name display
        cardName.addEventListener('input', (e) => {
            displayName.textContent = e.target.value.toUpperCase() || 'FULL NAME';
        });

        // Expiry date formatting and display
        expiry.addEventListener('input', (e) => {
            let value = e.target.value.replace(/\D/g, '');
            if (value.length >= 2) {
                value = value.slice(0, 2) + '/' + value.slice(2, 4);
            }
            e.target.value = value;
            displayExpiry.textContent = value || 'MM/YY';
        });

        // CVV - flip card when focused
        cvv.addEventListener('focus', () => {
            card.classList.add('flipped');
        });

        cvv.addEventListener('blur', () => {
            card.classList.remove('flipped');
        });

        cvv.addEventListener('input', (e) => {
            displayCVV.textContent = e.target.value || '***';
        });

        // Only allow numbers in card number and CVV
        cardNumber.addEventListener('keypress', (e) => {
            if (!/[0-9]/.test(e.key)) e.preventDefault();
        });

        cvv.addEventListener('keypress', (e) => {
            if (!/[0-9]/.test(e.key)) e.preventDefault();
        });

        // Form submission
        document.getElementById('cardForm').addEventListener('submit', (e)m => {
            e.preventDefault();
            alert('Payment processing... (Demo only)');
        });
    </script>
